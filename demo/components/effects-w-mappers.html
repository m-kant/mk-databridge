JavaScript

roundControl = function(val,mode){
  if (!/^\d+$/.test(val))throw new Error('value must be a number');
  if(mode === 'set'){
    $(this).siblings( '.pointer' ).css('transform','rotate('+val+'deg)');
  }
  return val;
};
